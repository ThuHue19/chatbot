TABLE: PAKH_LUU_LUONG_DICH_VU
DESCRIPTION: Sự cố ảnh hưởng đến lưu lượng dịch vụ, theo NE. 
COLUMNS:
- ID NUMBER NOT NULL ENABLE, 
- NGAY VARCHAR2(15 BYTE), 
- START_TIME DATE, 
-END_TIME DATE, 
- NE VARCHAR2(255 BYTE), DESTINATION VARCHAR2(255 BYTE), KET_NOI VARCHAR2(255 BYTE): điểm ảnh hưởng.
- SEVERITY VARCHAR2(50 BYTE), 
- ALARM_NAME VARCHAR2(255 BYTE), ALARM_INFO VARCHAR2(255 BYTE),		REGION VARCHAR2(100 BYTE), ALARM_TYPE VARCHAR2(100 BYTE):  thông tin lỗi.
- DEPT VARCHAR2(255 BYTE), TEAM VARCHAR2(255 BYTE), NETWORK VARCHAR2(255 BYTE): thông tin địa lý và tổ chức.
- AFFECTING_YOUTUBE VARCHAR2(10 BYTE), AFFECTING_FACEBOOK VARCHAR2(10 BYTE), AFFECTING_TIKTOK VARCHAR2(10 BYTE), AFFECTING_INSTAGRAM VARCHAR2(10 BYTE), AFFECTING_MESSENGER VARCHAR2(10 BYTE), AFFECTING_ZALO VARCHAR2(10 BYTE), AFFECTING_FPT_PLAY VARCHAR2(10 BYTE), AFFECTING_GOOGLE_PLAY VARCHAR2(10 BYTE), AFFECTING_OTHER_SERVICES VARCHAR2(10 BYTE): ột Boolean ảnh hưởng dịch vụ OTT 
- DURATION VARCHAR2(200 BYTE): thời lượng ảnh hưởn
- CREATE_DATE DATE DEFAULT SYSDATE, 
- UPDATED_DATE DATE
