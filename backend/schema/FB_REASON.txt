TABLE FB_REASON 
   (	ID NUMBER NOT NULL ENABLE, 
	NAME VARCHAR2(250 BYTE), 
	IS_ENABLE VARCHAR2(1 BYTE) DEFAULT 'Y', 
	PARENT_ID NUMBER DEFAULT 0, 
	ORDERING NUMBER(4,0), 
	CREATED_BY VARCHAR2(50 BYTE) DEFAULT 'admin', 
	CREATE_DATE DATE DEFAULT sysdate, 
	MODIFIED_BY VARCHAR2(50 BYTE), 
	MODIFY_DATE DATE, 
	FB_GROUP_ID NUMBER, 
	SLA_REGION VARCHAR2(20 BYTE), 
	REASON_GROUP VARCHAR2(200 BYTE)
   )
